\section{The Problem}

\begin{frame}{The Material Takeoff Challenge}
\begin{columns}
\column{0.5\textwidth}
\textbf{Current State:}
\begin{itemize}
    \item Estimators spend \textbf{50--80\%} of time on quantity takeoffs
    \item Manual counting leads to costly errors
    \item Inconsistent results across estimators
    \item Junior staff lack code/methodology knowledge
\end{itemize}

\column{0.5\textwidth}
\textbf{The Cost:}
\begin{itemize}
    \item Lumber waste: \textbf{15--20\%} typical
    \item Change orders from takeoff errors
    \item Limited bid capacity
    \item Lost institutional knowledge
\end{itemize}
\end{columns}

\vspace{1em}
\centering
\textit{``The construction industry has 0\% productivity growth over 20 years''} -- McKinsey
\end{frame}

\begin{frame}{What Estimators Need}
\begin{enumerate}
    \item \textbf{Accurate Bill of Materials (BOM)} from plans
    \item \textbf{Optimized cut lists} that minimize waste
    \item \textbf{Build instructions} aligned with codes
    \item \textbf{Machine-readable exports} for suppliers
    \item \textbf{Audit trail} for every decision
\end{enumerate}

\vspace{1em}
\begin{block}{Goal}
Transform plans into actionable, compliant, waste-minimized outputs---automatically.
\end{block}
\end{frame}

\begin{frame}{Why a Knowledge Graph?}
\begin{columns}
\column{0.55\textwidth}
\textbf{What the KG stores:}
\begin{itemize}
    \item Component definitions (studs, headers, assemblies)
    \item Material catalog (SKUs, stock lengths, grades)
    \item Fasteners \& connectors with rules
    \item Building codes (IRC/IBC by jurisdiction)
    \item Historical project outcomes
\end{itemize}

\column{0.45\textwidth}
\textbf{Benefits:}
\begin{itemize}
    \item \textbf{Auditable}: Every output traceable to rules
    \item \textbf{Updatable}: Change codes without code changes
    \item \textbf{Learning}: Improve from real outcomes
    \item \textbf{Reusable}: Apply learnings across projects
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Agent Responsibilities}
\begin{description}
    \item[Extraction QA] Validates geometry, flags low-confidence dimensions, requests reprocessing
    \item[Component Inference] Maps plan facts to assemblies (wall $\rightarrow$ studs, opening $\rightarrow$ header)
    \item[Code \& Compliance] Checks against IRC/IBC rules, produces citations
    \item[Procurement \& Cut] Selects stock lengths, calls optimizer, minimizes waste
    \item[Instruction Generation] Produces step-by-step build guides with code callouts
\end{description}
\end{frame}

\begin{frame}{Cut Optimization}
\begin{columns}
\column{0.5\textwidth}
\textbf{The Problem:}
\begin{itemize}
    \item NP-hard cutting stock problem
    \item Multiple stock lengths available
    \item Kerf (blade width) matters
    \item Bundling for crew efficiency
\end{itemize}

\column{0.5\textwidth}
\textbf{Our Approach:}
\begin{itemize}
    \item First-Fit Decreasing heuristic
    \item Integer Linear Programming (OR-Tools)
    \item Multi-stock optimization
    \item Substitution rules from KG
\end{itemize}
\end{columns}

\vspace{1em}
\begin{block}{Objective}
$\min \sum_{s \in S} c_s \cdot x_s + \lambda \cdot \text{waste}$
\end{block}
\end{frame}

\begin{frame}{Knowledge Graph \& Agents}
\begin{columns}
\column{0.5\textwidth}
\textbf{Graph Stack:}
\begin{itemize}
    \item Neo4j graph database
    \item Cypher query language
    \item Vector embeddings for semantic search
    \item Async Python driver
\end{itemize}

\column{0.5\textwidth}
\textbf{Agent Stack:}
\begin{itemize}
    \item Claude / GPT-4 LLMs
    \item LangChain orchestration
    \item ReAct reasoning patterns
    \item Structured tool calling
\end{itemize}
\end{columns}
\end{frame}
